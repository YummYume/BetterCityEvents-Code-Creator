# The Fall of Warsaw
news_event = {
	id = bettercitynews.2
	title = bettercitynews.2.t
	desc = bettercitynews.2.d
	picture = GFX_bettercityevents_fall_warsaw
	
	major = yes
	
	is_triggered_only = yes
	
	option = { #CURRENT OWNER
		name = bettercitynews.2.a
		trigger = {
			tag = event_target:bettercitynews_fall_of_warsaw_owner
            NOT = {
                tag = event_target:bettercitynews_fall_of_warsaw_controller
            }
		}
	}
	option = { #CURRENT CONTROLLER
		name = bettercitynews.2.b
		trigger = {
            tag = event_target:bettercitynews_fall_of_warsaw_controller
            NOT = {
                tag = event_target:bettercitynews_fall_of_warsaw_owner
            }
        }
	}
	option = { #NOT OWNER NOR CONTROLLER
		name = bettercitynews.2.c
		trigger = {
            NOT = {
            tag = event_target:bettercitynews_fall_of_warsaw_owner
				tag = event_target:bettercitynews_fall_of_warsaw_controller
            }
        }
	}
}
